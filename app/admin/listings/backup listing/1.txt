"use client";

import Link from "next/link";
import { useEffect, useMemo, useState } from "react";
import { supabase } from "../../../lib/supabaseClient";

type ListingRow = {
  id: string;
  created_at: string | null;
  user_id: string | null;
  title: string;
  description: string;
  category: string;
  price_usd: number | null;
  image_url: string | null;
  is_active: boolean | null;
  moderation_status: "pending" | "verified" | "suspended" | null;
  moderation_note: string | null;
  verified_at: string | null;
  last_reviewed_at: string | null;
};

type FilterStatus = "all" | "pending" | "verified" | "suspended";

function formatGs(value: number | string | null | undefined) {
  const n = Number(value ?? 0);
  return `Gs. ${n.toLocaleString("es-PY")}`;
}

function formatDateTime(value?: string | null) {
  if (!value) return "-";
  return new Date(value).toLocaleString();
}

function moderationLabel(status?: string | null) {
  switch (status) {
    case "pending":
      return "Pendiente";
    case "verified":
      return "Verificada";
    case "suspended":
      return "Suspendida";
    default:
      return "Sin estado";
  }
}

function moderationBadgeClass(status?: string | null) {
  switch (status) {
    case "pending":
      return "bg-yellow-400 text-black";
    case "verified":
      return "bg-[#2FA84F] text-black";
    case "suspended":
      return "bg-red-500 text-white";
    default:
      return "bg-white/10 text-white";
  }
}

function cardBorderClass(status?: string | null) {
  switch (status) {
    case "pending":
      return "border-yellow-400/70";
    case "verified":
      return "border-[#2FA84F]/70";
    case "suspended":
      return "border-red-500/70";
    default:
      return "border-white/10";
  }
}

export default function AdminListingsPage() {
  const [items, setItems] = useState<ListingRow[]>([]);
  const [msg, setMsg] = useState("Cargando publicaciones...");
  const [loading, setLoading] = useState(true);
  const [filterStatus, setFilterStatus] = useState<FilterStatus>("all");
  const [search, setSearch] = useState("");
  const [openId, setOpenId] = useState<string | null>(null);
  const [noteDrafts, setNoteDrafts] = useState<Record<string, string>>({});
  const [updatingId, setUpdatingId] = useState<string | null>(null);

  useEffect(() => {
    loadListings();
  }, []);

  async function loadListings() {
    setLoading(true);
    setMsg("Cargando publicaciones...");

    const { data, error } = await supabase
      .from("listings")
      .select(
        "id,created_at,user_id,title,description,category,price_usd,image_url,is_active,moderation_status,moderation_note,verified_at,last_reviewed_at"
      )
      .order("created_at", { ascending: false });

    setLoading(false);

    if (error) {
      setItems([]);
      setMsg("Error cargando publicaciones: " + error.message);
      return;
    }

    const rows = (data ?? []) as ListingRow[];
    setItems(rows);

    const initialDrafts: Record<string, string> = {};
    for (const row of rows) {
      initialDrafts[row.id] = row.moderation_note ?? "";
    }
    setNoteDrafts(initialDrafts);

    setMsg("");
  }

  async function logout() {
    await supabase.auth.signOut();
    window.location.href = "/";
  }

  function updateDraft(id: string, value: string) {
    setNoteDrafts((prev) => ({
      ...prev,
      [id]: value,
    }));
  }

  async function setModerationStatus(
    item: ListingRow,
    nextStatus: "pending" | "verified" | "suspended"
  ) {
    const note = (noteDrafts[item.id] ?? "").trim();

    if (nextStatus === "suspended" && !note) {
      alert(
        "Antes de suspender, escribe un motivo en la nota de moderacion.\n\nEjemplo: contacto en imagen, contacto en descripcion, corregir publicacion, etc."
      );
      return;
    }

    if (nextStatus === "verified") {
      const ok = window.confirm(
        `¿Verificar y activar la publicacion "${item.title}"?`
      );
      if (!ok) return;
    }

    if (nextStatus === "suspended") {
      const ok = window.confirm(
        `¿Suspender la publicacion "${item.title}"?`
      );
      if (!ok) return;
    }

    if (nextStatus === "pending") {
      const ok = window.confirm(
        `¿Volver a poner en pendiente la publicacion "${item.title}"?`
      );
      if (!ok) return;
    }

    setUpdatingId(item.id);

    const updateData: {
      moderation_status: "pending" | "verified" | "suspended";
      moderation_note: string | null;
      is_active: boolean;
      last_reviewed_at: string;
      verified_at?: string | null;
    } = {
      moderation_status: nextStatus,
      moderation_note: note || null,
      is_active: nextStatus === "verified",
      last_reviewed_at: new Date().toISOString(),
    };

    if (nextStatus === "verified") {
      updateData.verified_at = new Date().toISOString();
    }

    if (nextStatus === "pending") {
      updateData.verified_at = null;
    }

    const { error } = await supabase
      .from("listings")
      .update(updateData)
      .eq("id", item.id);

    setUpdatingId(null);

    if (error) {
      alert("Error actualizando publicacion: " + error.message);
      return;
    }

    await loadListings();
  }

  const filtered = useMemo(() => {
    const term = search.trim().toLowerCase();

    return items.filter((item) => {
      const statusOk =
        filterStatus === "all" ? true : item.moderation_status === filterStatus;

      if (!statusOk) return false;

      if (!term) return true;

      return (
        item.title?.toLowerCase().includes(term) ||
        item.description?.toLowerCase().includes(term) ||
        item.category?.toLowerCase().includes(term) ||
        (item.user_id ?? "").toLowerCase().includes(term)
      );
    });
  }, [items, filterStatus, search]);

  const stats = useMemo(() => {
    const pending = items.filter((x) => x.moderation_status === "pending").length;
    const verified = items.filter((x) => x.moderation_status === "verified").length;
    const suspended = items.filter((x) => x.moderation_status === "suspended").length;

    return {
      total: items.length,
      pending,
      verified,
      suspended,
    };
  }, [items]);

  return (
    <main className="min-h-screen bg-[#0b0b0b] px-4 py-10 text-white">
      <div className="mx-auto max-w-7xl">
        <div className="rounded-3xl border-2 border-green-500 bg-[#121212] p-6 shadow-[0_12px_45px_rgba(0,0,0,0.45)]">
          <div className="flex flex-col gap-3 md:flex-row md:items-center md:justify-between">
            <div>
              <h1 className="text-3xl font-black">
                Publicaciones de <span className="text-[#f4c400]">vendedores</span>
              </h1>
              <p className="mt-2 text-white/70">
                Aqui revisas publicaciones, imagenes, notas de moderacion y estados.
              </p>
            </div>

            <div className="flex flex-wrap gap-2">
              <button
                onClick={loadListings}
                className="rounded-2xl bg-white/10 px-4 py-2 font-semibold transition hover:bg-white/20 active:scale-[0.98]"
              >
                Recargar
              </button>

              <Link
                href="/admin"
                className="rounded-2xl bg-white/10 px-4 py-2 font-semibold transition hover:bg-white/20 active:scale-[0.98]"
              >
                Volver al admin
              </Link>

              <Link
                href="/"
                className="rounded-2xl bg-white/10 px-4 py-2 font-semibold transition hover:bg-white/20 active:scale-[0.98]"
              >
                Inicio
              </Link>

              <button
                onClick={logout}
                className="rounded-2xl bg-[#2FA84F] px-4 py-2 font-bold text-black transition hover:brightness-110 active:scale-[0.98]"
              >
                Cerrar sesion
              </button>
            </div>
          </div>

          {msg && (
            <div className="mt-5 rounded-2xl border border-white/10 bg-white/5 px-4 py-3 text-sm text-white shadow-md">
              {msg}
            </div>
          )}
        </div>

        <div className="mt-8 grid grid-cols-1 gap-4 md:grid-cols-2 xl:grid-cols-4">
          <div className="rounded-3xl border-2 border-white/20 bg-[#121212] p-5">
            <p className="text-sm text-white/60">Total publicaciones</p>
            <p className="mt-2 text-3xl font-black text-white">{stats.total}</p>
          </div>

          <div className="rounded-3xl border-2 border-yellow-400/60 bg-[#121212] p-5">
            <p className="text-sm text-white/60">Pendientes</p>
            <p className="mt-2 text-3xl font-black text-yellow-400">{stats.pending}</p>
          </div>

          <div className="rounded-3xl border-2 border-[#2FA84F]/60 bg-[#121212] p-5">
            <p className="text-sm text-white/60">Verificadas</p>
            <p className="mt-2 text-3xl font-black text-[#2FA84F]">{stats.verified}</p>
          </div>

          <div className="rounded-3xl border-2 border-red-500/60 bg-[#121212] p-5">
            <p className="text-sm text-white/60">Suspendidas</p>
            <p className="mt-2 text-3xl font-black text-red-500">{stats.suspended}</p>
          </div>
        </div>

        <div className="mt-8 rounded-3xl border border-white/10 bg-[#121212] p-5 shadow-[0_12px_45px_rgba(0,0,0,0.45)]">
          <div className="flex flex-col gap-3 lg:flex-row lg:items-center lg:justify-between">
            <div className="flex flex-wrap gap-2">
              <button
                onClick={() => setFilterStatus("all")}
                className={`rounded-2xl px-4 py-2 text-sm font-bold transition ${
                  filterStatus === "all"
                    ? "bg-white text-black"
                    : "bg-white/10 text-white hover:bg-white/20"
                }`}
              >
                Todas
              </button>

              <button
                onClick={() => setFilterStatus("pending")}
                className={`rounded-2xl px-4 py-2 text-sm font-bold transition ${
                  filterStatus === "pending"
                    ? "bg-yellow-400 text-black"
                    : "bg-white/10 text-white hover:bg-white/20"
                }`}
              >
                Pendientes
              </button>

              <button
                onClick={() => setFilterStatus("verified")}
                className={`rounded-2xl px-4 py-2 text-sm font-bold transition ${
                  filterStatus === "verified"
                    ? "bg-[#2FA84F] text-black"
                    : "bg-white/10 text-white hover:bg-white/20"
                }`}
              >
                Verificadas
              </button>

              <button
                onClick={() => setFilterStatus("suspended")}
                className={`rounded-2xl px-4 py-2 text-sm font-bold transition ${
                  filterStatus === "suspended"
                    ? "bg-red-500 text-white"
                    : "bg-white/10 text-white hover:bg-white/20"
                }`}
              >
                Suspendidas
              </button>
            </div>

            <input
              value={search}
              onChange={(e) => setSearch(e.target.value)}
              placeholder="Buscar por titulo, descripcion, categoria o user id..."
              className="w-full rounded-2xl bg-white/5 px-4 py-3 text-sm outline-none ring-1 ring-white/10 lg:max-w-xl"
            />
          </div>
        </div>

        {loading ? (
          <div className="mt-8 rounded-3xl border border-white/10 bg-[#121212] p-6 text-white/70">
            Cargando publicaciones...
          </div>
        ) : filtered.length === 0 ? (
          <div className="mt-8 rounded-3xl border border-white/10 bg-[#121212] p-6 text-white/70">
            No hay publicaciones para mostrar con ese filtro.
          </div>
        ) : (
          <div className="mt-8 grid grid-cols-1 gap-4">
            {filtered.map((item) => {
              const isOpen = openId === item.id;

              return (
                <div
                  key={item.id}
                  className={`rounded-3xl border-2 bg-[#121212] p-5 shadow-[0_12px_45px_rgba(0,0,0,0.35)] ${cardBorderClass(
                    item.moderation_status
                  )}`}
                >
                  <div className="flex flex-col gap-4 xl:flex-row xl:items-start xl:justify-between">
                    <div className="flex-1">
                      <div className="flex flex-wrap items-center gap-3">
                        <span
                          className={`rounded-full px-3 py-1 text-xs font-black ${moderationBadgeClass(
                            item.moderation_status
                          )}`}
                        >
                          {moderationLabel(item.moderation_status)}
                        </span>

                        <span
                          className={`rounded-full px-3 py-1 text-xs font-bold ${
                            item.is_active
                              ? "bg-[#2FA84F]/20 text-[#2FA84F]"
                              : "bg-red-500/20 text-red-300"
                          }`}
                        >
                          {item.is_active ? "Activa" : "Inactiva"}
                        </span>
                      </div>

                      <h2 className="mt-3 text-xl font-black">{item.title}</h2>

                      <div className="mt-3 grid grid-cols-1 gap-3 md:grid-cols-2 xl:grid-cols-4">
                        <div className="rounded-2xl bg-white/5 p-4 ring-1 ring-white/10">
                          <p className="text-xs text-white/50">Categoria</p>
                          <p className="mt-1 font-semibold text-white">{item.category}</p>
                        </div>

                        <div className="rounded-2xl bg-white/5 p-4 ring-1 ring-white/10">
                          <p className="text-xs text-white/50">Precio</p>
                          <p className="mt-1 font-semibold text-[#f4c400]">
                            {formatGs(item.price_usd)}
                          </p>
                        </div>

                        <div className="rounded-2xl bg-white/5 p-4 ring-1 ring-white/10">
                          <p className="text-xs text-white/50">Creado</p>
                          <p className="mt-1 font-semibold text-white">
                            {formatDateTime(item.created_at)}
                          </p>
                        </div>

                        <div className="rounded-2xl bg-white/5 p-4 ring-1 ring-white/10">
                          <p className="text-xs text-white/50">User ID</p>
                          <p className="mt-1 break-all font-semibold text-white">
                            {item.user_id || "-"}
                          </p>
                        </div>
                      </div>
                    </div>

                    <div className="flex flex-wrap gap-2 xl:w-[340px] xl:justify-end">
                      {item.moderation_status !== "verified" && (
                        <button
                          onClick={() => setModerationStatus(item, "verified")}
                          disabled={updatingId === item.id}
                          className="rounded-2xl bg-[#2FA84F] px-4 py-2 text-sm font-bold text-black transition hover:brightness-110 disabled:opacity-60"
                        >
                          {updatingId === item.id ? "Guardando..." : "Verificar"}
                        </button>
                      )}

                      {item.moderation_status !== "suspended" && (
                        <button
                          onClick={() => setModerationStatus(item, "suspended")}
                          disabled={updatingId === item.id}
                          className="rounded-2xl bg-red-500 px-4 py-2 text-sm font-bold text-white transition hover:brightness-110 disabled:opacity-60"
                        >
                          {updatingId === item.id ? "Guardando..." : "Suspender"}
                        </button>
                      )}

                      {item.moderation_status === "suspended" && (
                        <button
                          onClick={() => setModerationStatus(item, "verified")}
                          disabled={updatingId === item.id}
                          className="rounded-2xl bg-[#f4c400] px-4 py-2 text-sm font-bold text-black transition hover:brightness-110 disabled:opacity-60"
                        >
                          {updatingId === item.id ? "Guardando..." : "Reactivar"}
                        </button>
                      )}

                      {item.moderation_status !== "pending" && (
                        <button
                          onClick={() => setModerationStatus(item, "pending")}
                          disabled={updatingId === item.id}
                          className="rounded-2xl bg-white/10 px-4 py-2 text-sm font-bold text-white transition hover:bg-white/20 disabled:opacity-60"
                        >
                          {updatingId === item.id ? "Guardando..." : "Poner pendiente"}
                        </button>
                      )}

                      <button
                        onClick={() => setOpenId(isOpen ? null : item.id)}
                        className="rounded-2xl bg-white/10 px-4 py-2 text-sm font-bold text-white transition hover:bg-white/20"
                      >
                        {isOpen ? "Ocultar detalle" : "Ver detalle"}
                      </button>
                    </div>
                  </div>

                  {isOpen && (
                    <div className="mt-5 grid grid-cols-1 gap-5 xl:grid-cols-[300px_1fr]">
                      <div className="overflow-hidden rounded-3xl border border-white/10 bg-[#0f0f0f]">
                        {item.image_url ? (
                          <img
                            src={item.image_url}
                            alt={item.title}
                            className="h-[280px] w-full object-contain bg-[#181818]"
                          />
                        ) : (
                          <div className="flex h-[280px] items-center justify-center text-sm text-white/50">
                            Sin imagen
                          </div>
                        )}
                      </div>

                      <div className="grid gap-4">
                        <div className="rounded-3xl border border-white/10 bg-white/5 p-5">
                          <p className="text-xs text-white/50">Descripcion</p>
                          <p className="mt-2 whitespace-pre-wrap text-white/85">
                            {item.description}
                          </p>
                        </div>

                        <div className="grid grid-cols-1 gap-4 md:grid-cols-2">
                          <div className="rounded-3xl border border-white/10 bg-white/5 p-5">
                            <p className="text-xs text-white/50">Verificada en</p>
                            <p className="mt-2 font-semibold text-white">
                              {formatDateTime(item.verified_at)}
                            </p>
                          </div>

                          <div className="rounded-3xl border border-white/10 bg-white/5 p-5">
                            <p className="text-xs text-white/50">Ultima revision</p>
                            <p className="mt-2 font-semibold text-white">
                              {formatDateTime(item.last_reviewed_at)}
                            </p>
                          </div>
                        </div>

                        <div className="rounded-3xl border border-[#f4c400]/20 bg-[#f4c400]/5 p-5">
                          <p className="text-sm font-bold text-[#f4c400]">
                            Nota de moderacion / motivo
                          </p>
                          <textarea
                            value={noteDrafts[item.id] ?? ""}
                            onChange={(e) => updateDraft(item.id, e.target.value)}
                            rows={4}
                            placeholder="Ejemplo: contacto en imagen, corregir descripcion, aprobado correctamente..."
                            className="mt-3 w-full resize-none rounded-2xl bg-black/20 px-4 py-3 text-sm text-white outline-none ring-1 ring-white/10 focus:ring-[#f4c400]/40"
                          />

                          <div className="mt-3 rounded-2xl border border-white/10 bg-black/20 px-4 py-3 text-sm text-white/70">
                            Si suspendes una publicacion, escribe aqui el motivo. Esa nota
                            luego servira para mostrar el aviso al vendedor.
                          </div>
                        </div>
                      </div>
                    </div>
                  )}
                </div>
              );
            })}
          </div>
        )}
      </div>
    </main>
  );
}
